i3 Window Manager



# Install
sudo apt install i3
# logout and log back in
# select i3 in login screen
# define mod_key - window/alt key


# Open new window
mod_key-enter

# Close window
mod_key-shift-e


# Tile vertically
mod_key-v

# Tile horizontally
mod_key-h


# Navigation keys
arrow keys
# or
mod_key-j           # left
mod_key-;           # right
mod_key-l           # up
mod_key-k           # down


mod_key-<number>            # go to workspace
mod_key-shift-<number>      # move current to the workspace <number>


mod_key-d           # menu
